import { useState } from 'react';

// Aula 68 (Ordem de execução dos Hooks)

// 'Lazy Initializers' é quando se passa uma 'callback' como
// valor inicial do 'hook' useState();
export const Home = () => {
  const [state, setState] = useState(() => {
    console.log('Hello, world!');
  });
};
// São executados ANTES da RENDERIZAÇÃO do componente.

// Um 'estado' no 'componente filho' não causa uma re-renderização
// no 'componente pai'.

// Os hooks do tipo 'useEffect' são executados NA ORDEM que o programador
// definir.
