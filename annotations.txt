// É possível "herdar" os estilos de um 'styled-component'
// criado:
import styled from 'styled-components';

export const Container = styled.div`
  background: ${({ theme }) => theme.colors.secondaryBg};
`;

export const ContainerTwo = styled(Container)`
  color: white;
`;
// Assim, o componente 'ContainerTwo' herda todas as
// propriedades css do 'Container' podendo alterar
// quaisquer 'css rules'.

// É possível SOBRESCREVER a tag gerada (styled.div) por
// outra:
export const ContainerThree = styled(Container).attrs({ as: 'article' })`
  color: white;
`;
// Ao inspecionar os elementos da DOM, o componente 'ContainerThree'
// é uma tag 'article' com as mesmas 'css rules' que o componente 'Container'.
