};

// 'html', é possível utilizar a unidade de medida relativa 'rem'
// como se fosse 'px', ou seja, os dev's normalmente informam
// '16px' ou '32px'. Com essas configurações, é possível utilizar
// '1.6rem' ou '3.2rem'.
